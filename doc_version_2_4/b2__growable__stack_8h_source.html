33template <typename T, int32 N>
34class b2GrowableStack
35{
36public:
37 b2GrowableStack()
38 {
39 m_stack = m_array;
40 m_count = 0;
41 m_capacity = N;
42 }
43
44 ~b2GrowableStack()
45 {
46 if (m_stack != m_array)
47 {
48 b2Free(m_stack);
49 m_stack = nullptr;
50 }
51 }
52
53 void Push(const T& element)
54 {
55 if (m_count == m_capacity)
56 {
57 T* old = m_stack;
58 m_capacity *= 2;
59 m_stack = (T*)b2Alloc(m_capacity * sizeof(T));
60 memcpy(m_stack, old, m_count * sizeof(T));
61 if (old != m_array)
62 {
63 b2Free(old);
64 }
65 }
66
67 m_stack[m_count] = element;
68 ++m_count;
69 }
70
71 T Pop()
72 {
73 b2Assert(m_count > 0);
74 --m_count;
75 return m_stack[m_count];
76 }
77
78 int32 GetCount()
79 {
80 return m_count;
81 }
82
83private:
84 T* m_stack;
85 T m_array[N];
86 int32 m_count;
87 int32 m_capacity;
88};

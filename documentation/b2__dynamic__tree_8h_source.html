189 template <typename T>
190 inline void b2DynamicTree::Query(T* callback, const b2AABB& aabb) const
191 {
192  b2GrowableStack<int32, 256> stack;
193  stack.Push(m_root);
194 
195  while (stack.GetCount() > 0)
196  {
197  int32 nodeId = stack.Pop();
198  if (nodeId == b2_nullNode)
199  {
200  continue;
201  }
202 
203  const b2TreeNode* node = m_nodes + nodeId;
204 
205  if (b2TestOverlap(node->aabb, aabb))
206  {
207  if (node->IsLeaf())
208  {
209  bool proceed = callback->QueryCallback(nodeId);
210  if (proceed == false)
211  {
212  return;
213  }
214  }
215  else
216  {
217  stack.Push(node->child1);
218  stack.Push(node->child2);
219  }
220  }
221  }
222 }

224 template <typename T>
225 inline void b2DynamicTree::RayCast(T* callback, const b2RayCastInput& input) const
226 {
227  b2Vec2 p1 = input.p1;
228  b2Vec2 p2 = input.p2;
229  b2Vec2 r = p2 - p1;
230  b2Assert(r.LengthSquared() > 0.0f);
231  r.Normalize();
232 
233  // v is perpendicular to the segment.
234  b2Vec2 v = b2Cross(1.0f, r);
235  b2Vec2 abs_v = b2Abs(v);
236 
237  // Separating axis for segment (Gino, p80).
238  // |dot(v, p1 - c)| > dot(|v|, h)
239 
240  float maxFraction = input.maxFraction;
241 
242  // Build a bounding box for the segment.
243  b2AABB segmentAABB;
244  {
245  b2Vec2 t = p1 + maxFraction * (p2 - p1);
246  segmentAABB.lowerBound = b2Min(p1, t);
247  segmentAABB.upperBound = b2Max(p1, t);
248  }
249 
250  b2GrowableStack<int32, 256> stack;
251  stack.Push(m_root);
252 
253  while (stack.GetCount() > 0)
254  {
255  int32 nodeId = stack.Pop();
256  if (nodeId == b2_nullNode)
257  {
258  continue;
259  }
260 
261  const b2TreeNode* node = m_nodes + nodeId;
262 
263  if (b2TestOverlap(node->aabb, segmentAABB) == false)
264  {
265  continue;
266  }
267 
268  // Separating axis for segment (Gino, p80).
269  // |dot(v, p1 - c)| > dot(|v|, h)
270  b2Vec2 c = node->aabb.GetCenter();
271  b2Vec2 h = node->aabb.GetExtents();
272  float separation = b2Abs(b2Dot(v, p1 - c)) - b2Dot(abs_v, h);
273  if (separation > 0.0f)
274  {
275  continue;
276  }
277 
278  if (node->IsLeaf())
279  {
280  b2RayCastInput subInput;
281  subInput.p1 = input.p1;
282  subInput.p2 = input.p2;
283  subInput.maxFraction = maxFraction;
284 
285  float value = callback->RayCastCallback(subInput, nodeId);
286 
287  if (value == 0.0f)
288  {
289  // The client has terminated the ray cast.
290  return;
291  }
292 
293  if (value > 0.0f)
294  {
295  // Update segment bounding box.
296  maxFraction = value;
297  b2Vec2 t = p1 + maxFraction * (p2 - p1);
298  segmentAABB.lowerBound = b2Min(p1, t);
299  segmentAABB.upperBound = b2Max(p1, t);
300  }
301  }
302  else
303  {
304  stack.Push(node->child1);
305  stack.Push(node->child2);
306  }
307  }
308 }

306B2_INLINE b2Rot b2IntegrateRotation( b2Rot q1, float deltaAngle )
307{
308 // dc/dt = -omega * sin(t)
309 // ds/dt = omega * cos(t)
310 // c2 = c1 - omega * h * s1
311 // s2 = s1 + omega * h * c1
312 b2Rot q2 = { q1.c - deltaAngle * q1.s, q1.s + deltaAngle * q1.c };
313 float mag = sqrtf( q2.s * q2.s + q2.c * q2.c );
314 float invMag = mag > 0.0 ? 1.0f / mag : 0.0f;
315 b2Rot qn = { q2.c * invMag, q2.s * invMag };
316 return qn;
317}

323B2_INLINE float b2ComputeAngularVelocity( b2Rot q1, b2Rot q2, float inv_h )
324{
325 // ds/dt = omega * cos(t)
326 // dc/dt = -omega * sin(t)
327 // s2 = s1 + omega * h * c1
328 // c2 = c1 - omega * h * s1
329
330 // omega * h * s1 = c1 - c2
331 // omega * h * c1 = s2 - s1
332 // omega * h = (c1 - c2) * s1 + (s2 - s1) * c1;
333 // omega * h = s1 * c1 - c2 * s1 + s2 * c1 - s1 * c1
334 // omega * h = s2 * c1 - c2 * s1 = sin(a2 - a1) ~= a2 - a1 for small delta
335 float omega = inv_h * ( q2.s * q1.c - q2.c * q1.s );
336 return omega;
337}

361B2_INLINE b2Rot b2MulRot( b2Rot q, b2Rot r )
362{
363 // [qc -qs] * [rc -rs] = [qc*rc-qs*rs -qc*rs-qs*rc]
364 // [qs qc] [rs rc] [qs*rc+qc*rs -qs*rs+qc*rc]
365 // s(q + r) = qs * rc + qc * rs
366 // c(q + r) = qc * rc - qs * rs
367 b2Rot qr;
368 qr.s = q.s * r.c + q.c * r.s;
369 qr.c = q.c * r.c - q.s * r.s;
370 return qr;
371}
372
374B2_INLINE b2Rot b2InvMulRot( b2Rot q, b2Rot r )
375{
376 // [ qc qs] * [rc -rs] = [qc*rc+qs*rs -qc*rs+qs*rc]
377 // [-qs qc] [rs rc] [-qs*rc+qc*rs qs*rs+qc*rc]
378 // s(q - r) = qc * rs - qs * rc
379 // c(q - r) = qc * rc + qs * rs
380 b2Rot qr;
381 qr.s = q.c * r.s - q.s * r.c;
382 qr.c = q.c * r.c + q.s * r.s;
383 return qr;
384}

387B2_INLINE float b2RelativeAngle( b2Rot b, b2Rot a )
388{
389 // sin(b - a) = bs * ac - bc * as
390 // cos(b - a) = bc * ac + bs * as
391 float s = b.s * a.c - b.c * a.s;
392 float c = b.c * a.c + b.s * a.s;
393 return atan2f( s, c );
394}

471B2_INLINE b2Mat22 b2GetInverse22( b2Mat22 A )
472{
473 float a = A.cx.x, b = A.cy.x, c = A.cx.y, d = A.cy.y;
474 float det = a * d - b * c;
475 if ( det != 0.0f )
476 {
477 det = 1.0f / det;
478 }
479
480 b2Mat22 B = {
481 { det * d, -det * c },
482 { -det * b, det * a },
483 };
484 return B;
485}

489B2_INLINE b2Vec2 b2Solve22( b2Mat22 A, b2Vec2 b )
490{
491 float a11 = A.cx.x, a12 = A.cy.x, a21 = A.cx.y, a22 = A.cy.y;
492 float det = a11 * a22 - a12 * a21;
493 if ( det != 0.0f )
494 {
495 det = 1.0f / det;
496 }
497 b2Vec2 x = { det * ( a22 * b.x - a12 * b.y ), det * ( a11 * b.y - a21 * b.x ) };
498 return x;
499}

136 inline bool b2PairLessThan(const b2Pair& pair1, const b2Pair& pair2)
137 {
138  if (pair1.proxyIdA < pair2.proxyIdA)
139  {
140  return true;
141  }
142 
143  if (pair1.proxyIdA == pair2.proxyIdA)
144  {
145  return pair1.proxyIdB < pair2.proxyIdB;
146  }
147 
148  return false;
149 }

188 template <typename T>
189 void b2BroadPhase::UpdatePairs(T* callback)
190 {
191  // Reset pair buffer
192  m_pairCount = 0;
193 
194  // Perform tree queries for all moving proxies.
195  for (int32 i = 0; i < m_moveCount; ++i)
196  {
197  m_queryProxyId = m_moveBuffer[i];
198  if (m_queryProxyId == e_nullProxy)
199  {
200  continue;
201  }
202 
203  // We have to query the tree with the fat AABB so that
204  // we don't fail to create a pair that may touch later.
205  const b2AABB& fatAABB = m_tree.GetFatAABB(m_queryProxyId);
206 
207  // Query tree, create pairs and add them pair buffer.
208  m_tree.Query(this, fatAABB);
209  }
210 
211  // Reset move buffer
212  m_moveCount = 0;
213 
214  // Sort the pair buffer to expose duplicates.
215  std::sort(m_pairBuffer, m_pairBuffer + m_pairCount, b2PairLessThan);
216 
217  // Send the pairs back to the client.
218  int32 i = 0;
219  while (i < m_pairCount)
220  {
221  b2Pair* primaryPair = m_pairBuffer + i;
222  void* userDataA = m_tree.GetUserData(primaryPair->proxyIdA);
223  void* userDataB = m_tree.GetUserData(primaryPair->proxyIdB);
224 
225  callback->AddPair(userDataA, userDataB);
226  ++i;
227 
228  // Skip any duplicate pairs.
229  while (i < m_pairCount)
230  {
231  b2Pair* pair = m_pairBuffer + i;
232  if (pair->proxyIdA != primaryPair->proxyIdA || pair->proxyIdB != primaryPair->proxyIdB)
233  {
234  break;
235  }
236  ++i;
237  }
238  }
239 
240  // Try to keep the tree balanced.
241  //m_tree.Rebalance(4);
242 }

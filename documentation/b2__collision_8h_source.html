207  {

258 inline bool b2AABB::IsValid() const
259 {
260  b2Vec2 d = upperBound - lowerBound;
261  bool valid = d.x >= 0.0f && d.y >= 0.0f;
262  valid = valid && lowerBound.IsValid() && upperBound.IsValid();
263  return valid;
264 }
265 
266 inline bool b2TestOverlap(const b2AABB& a, const b2AABB& b)
267 {
268  b2Vec2 d1, d2;
269  d1 = b.lowerBound - a.upperBound;
270  d2 = a.lowerBound - b.upperBound;
271 
272  if (d1.x > 0.0f || d1.y > 0.0f)
273  return false;
274 
275  if (d2.x > 0.0f || d2.y > 0.0f)
276  return false;
277 
278  return true;
279 }

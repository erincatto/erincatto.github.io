135 inline int32 b2DistanceProxy::GetSupport(const b2Vec2& d) const
136 {
137  int32 bestIndex = 0;
138  float bestValue = b2Dot(m_vertices[0], d);
139  for (int32 i = 1; i < m_count; ++i)
140  {
141  float value = b2Dot(m_vertices[i], d);
142  if (value > bestValue)
143  {
144  bestIndex = i;
145  bestValue = value;
146  }
147  }
148 
149  return bestIndex;
150 }
